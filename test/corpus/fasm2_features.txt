; FASM2 style macro with lambda
macro times? counter, statement&
    repeat counter
        statement
    end repeat
end macro

times 10, mov [edi], eax
        add edi, 4

; FASM2 style block macro
macro struct? name
{
    virtual at 0
    name:
}

macro ends?
{
    end virtual
}

struct POINT
    .x dd ?
    .y dd ?
ends

    end match
